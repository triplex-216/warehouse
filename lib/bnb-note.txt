    branch(node_src ): 
        global_queue = PriorityQueue
        route = []

        while global_queue: 
            pop node_src from global_queue

            for each ap_src in node_src
                for each reachable ap_dest from ap_src: 
                    // Create data copies
                    route_copy <- route
                    mat_copy <- mat

                    add ap_dest to route_copy
                    cost <- reduce(mat_copy)
                    add (weight=cost, route_copy, mat_copy) to global_queue

        // all reachable APs have been explored
